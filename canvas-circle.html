Python inside HTML: Drawing circles on click

from pyodide import to_js from js import ImageData, Uint8ClampedArray, window import random; from math import pi canvas=Element("my-canvas").element canvas.width=window.innerWidth; canvas.height=500 ctx = canvas.getContext("2d") ctx.clearRect(0, 0, canvas.width, canvas.height) def drawCircle(*args, **kwargs): width,height=canvas.width,canvas.height x=width*random.random(); y=height*random.random() radius=7+.1*width*random.random() r,g,b=random.randint(0,255),random.randint(0,255),random.randint(0,255) ctx.beginPath(); ctx.arc(x, y, radius, 0, 2 * pi) ctx.fillStyle = f"rgba({r},{g},{b}, 0.5)"; ctx.fill()
